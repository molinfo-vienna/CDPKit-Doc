5.1.1. Extended Connectivity Fingerprints (ECFPs)ÔÉÅ

 1import CDPL.Chem as Chem
 2import CDPL.Descr as Descr
 3import CDPL.Util as Util
 4
 5
 6# generates the binary ECFP for the given molecule
 7def genECFP(mol: Chem.Molecule, num_bits: int, radius: int, inc_hs: bool, inc_config: bool) -> Util.BitSet:
 8    Chem.calcImplicitHydrogenCounts(mol, False)        # calculate implicit hydrogen counts (if not yet done)
 9    Chem.perceiveHybridizationStates(mol, False)       # perceive atom hybridization states and set corresponding property for all atoms
10    Chem.setRingFlags(mol, False)                      # perceive cycles and set corresponding atom and bond properties
11    Chem.perceiveSSSR(mol, False)                      # perceive smallest set of smallest rings and store as Chem.MolecularGraph property
12    Chem.setAromaticityFlags(mol, False)               # perceive aromaticity and set corresponding atom and bond properties
13    
14    ecfp_gen = Descr.CircularFingerprintGenerator()    # create ECFP generator instance
15
16    if inc_config:
17        ecfp_gen.includeChirality(True)                # allow atom chirality to have an impact on the ECFP generation
18        Chem.calcAtomStereoDescriptors(mol, False)     # calculate atom stereo descriptors and set corresponding property for all atoms
19
20    if inc_hs:        
21        ecfp_gen.includeHydrogens(True)                # include explicit hydrogens in the ECFP generation
22        Chem.makeHydrogenComplete(mol)                 # make any implicit hydrogens explicit
23         
24    fp = Util.BitSet()                                 # create fingerprint bitset
25    fp.resize(num_bits)                                # set desired fingerprint size
26
27    ecfp_gen.setNumIterations(radius)                  # set num. iterations (=atom. env. radius)
28    ecfp_gen.generate(mol)                             # extract chracteristic structural features
29    ecfp_gen.setFeatureBits(fp)                        # set bits associated with the extracted structural features
30
31    # if needed, fp could be converted into a numpy single precision float array as follows:
32    # fp = numpy.array(fp, dtype=numpy.float32)
33    
34    return fp
