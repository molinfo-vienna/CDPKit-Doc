4.2.2. Receptor Binding Pocket Descriptor

The script gen_kuvek_bp_descr.py generates and outputs a binding pocket shape/electrostatics descriptor for a given receptor structure according to the procedure devised by Kuvek et al.

Synopsis

python gen_kuvek_bp_descr.py [-h] -i <file> -o <file> -c <float> <float> <float> [-r <float>] [-s <three-letter code> [<three-letter code> …]] [-n <int>] [-t] [-q]

Mandatory options

-i <file>

Receptor structure input file (.mol2, .pdb, .mmtf, .cif, .mmcif)

-o <file>

Descriptor output file

Other options

-h, --help

Show help message and exit

-n <integer>

Number of intersection test vectors (default: 492)

-r <float>

Probe sphere radius (default: 20.0)

-s <three-letter code> [<three-letter code> …]

Whitespace separated list of PDB three-letter codes specifying residues to remove from the receptor structure (e.g. an existing ligand)

-t

Output test vector xyz values for each descriptor element (default: false)

-q

Disable progress output (default: false)

  7import CDPL.Descr as Descr
  8import CDPL.Math as Math

 11# reads and preprocesses the specified receptor structure
 12def readAndPrepareReceptorStructure(args: argparse.Namespace) -> Chem.Molecule:
 13    sup_fmts = [ Chem.DataFormat.MOL2,
 14                 Biomol.DataFormat.PDB,
 15                 Biomol.DataFormat.MMTF,
 16                 Biomol.DataFormat.MMCIF ]
 17
 18    # create reader for receptor structure (format specified by file extension)
 19    reader = Chem.MoleculeReader(args.receptor_file) 
 20                           
 21    if reader.getDataFormat() not in sup_fmts:   # check if the format is supported by this script
 22        sys.exit('Error: receptor input file format not supported')
 23    
 24    rec_mol = Chem.BasicMolecule()    # create an instance of the default implementation of the
 25                                      # Chem.Molecule interface that will store the receptor struct.
 26    try:
 27        if not reader.read(rec_mol):  # read receptor structure
 28            sys.exit('Error: reading receptor structure failed')
 29
 30    except Exception as e:
 31        sys.exit(f'Error: reading receptor structure failed:\n{str(e)}')            
 32
 33    # preprocess the receptor structure (removal of residues and
 34    # calculation of properties required by the pharm. generation procedure)
 35    try:
 36        rem_atoms = False
 37
 38        # delete atoms belonging to residues that should be stripped
 39        if args.strip_res_list:            
 40            atoms_to_rem = Chem.Fragment() # will store the atoms to delete
 41            res_to_strip = { tlc.upper() for tlc in args.strip_res_list }
 42        
 43            for atom in rec_mol.atoms:     # identify and note atoms belonging to the stripped residues
 44                if Biomol.getResidueCode(atom).upper() in res_to_strip:
 45                    atoms_to_rem.addAtom(atom)
 46
 47            if atoms_to_rem.numAtoms > 0:
 48                rec_mol -= atoms_to_rem    # delete atoms from the receptor structure
 49                rem_atoms = True
 50
 51                if not args.quiet:
 52                    print(f'! Removed {atoms_to_rem.numAtoms} atom(s) from the receptor structure')
 53
 54        # calc required receptor structure properties
 55        Chem.perceiveSSSR(rec_mol, rem_atoms)
 56        Chem.setRingFlags(rec_mol, rem_atoms)
 57        Chem.calcImplicitHydrogenCounts(rec_mol, rem_atoms)
 58        Chem.perceiveHybridizationStates(rec_mol, rem_atoms)
 59        Chem.setAromaticityFlags(rec_mol, rem_atoms)
 60
 61        if Chem.makeHydrogenComplete(rec_mol):                    # make implicit hydrogens (if any) explicit
 62            Chem.calcHydrogen3DCoordinates(rec_mol)               # calculate 3D coordinates for the added expl. hydrogens
 63            Biomol.setHydrogenResidueSequenceInfo(rec_mol, False) # set residue information for the added expl. hydrogens
 64
 65        # calc MMFF94 partial charges
 66        ForceField.perceiveMMFF94AromaticRings(rec_mol, False)        # perceive aromatic rings according to the MMFF94 aroamticity model and store data as Chem.MolecularGraph property
 67        ForceField.assignMMFF94AtomTypes(rec_mol, False, False)       # perceive MMFF94 atom types (tolerant mode) set corresponding property for all atoms
 68        ForceField.assignMMFF94BondTypeIndices(rec_mol, False, False) # perceive MMFF94 bond types (tolerant mode) set corresponding property for all bonds
 69        ForceField.calcMMFF94AtomCharges(rec_mol, False, False)       # calculate MMFF94 atom charges (tolerant mode) set corresponding property for all atoms
 70
 71    except Exception as e:
 72        sys.exit(f'Error: processing of receptor structure failed:\n{str(e)}')            
 73
 74    return rec_mol

 76def parseArgs() -> argparse.Namespace:
 77    parser = argparse.ArgumentParser(description='Generates a binding pocket shape/electrostatics descriptor for a given receptor '\
 78                                    'structure according to the procedure devised by Kuvek et al.')
 79
 80    parser.add_argument('-i',
 81                        dest='receptor_file',
 82                        required=True,
 83                        metavar='<file>',
 84                        help='Receptor structure input file (*.mol2, *.pdb, *.mmtf, *.cif, *.mmcif)')
 85    parser.add_argument('-o',
 86                        dest='out_file',
 87                        required=True,
 88                        metavar='<file>',
 89                        help='Descriptor output file')
 90    parser.add_argument('-s',
 91                        dest='strip_res_list',
 92                        required=False,
 93                        metavar='<three-letter code>',
 94                        nargs='+',
 95                        help='Whitespace separated list of PDB three-letter codes specifying residues to remove from the receptor structure (e.g. an existing ligand)')
 96    parser.add_argument('-c',
 97                        dest='sphere_ctr',
 98                        required=True,
 99                        metavar='<float>',
100                        nargs='+',
101                        type=float,
102                        help='Whitespace separated list of probe sphere center x, y and z coordinates')
103    parser.add_argument('-r',
104                        dest='sphere_rad',
105                        required=False,
106                        metavar='<float>',
107                        type=float,
108                        default=20.0,
109                        help='Probe sphere radius (default: 20.0)')
110    parser.add_argument('-n',
111                        dest='num_test_vecs',
112                        required=False,
113                        metavar='<integer>',
114                        type=int,
115                        default=492,
116                        help='Number of intersection test vectors (default: 492)')
117    parser.add_argument('-t',
118                        dest='write_test_vecs',
119                        required=False,
120                        action='store_true',
121                        default=False,
122                        help='Output test vector xyz values for each descriptor element (default: false)')
123    parser.add_argument('-q',
124                        dest='quiet',
125                        required=False,
126                        action='store_true',
127                        default=False,
128                        help='Disable progress output (default: false)')
129
130    return parser.parse_args()

132def main() -> None:
133    args = parseArgs()
134
135    if len(args.sphere_ctr) != 3:
136        sys.exit('Error: sphere center coordinates missing or incomplete')       
137    
138    # read and preprocess the receptor structure
139    rec_mol = readAndPrepareReceptorStructure(args)
140
141    # create and initialize an instance of the class Descr.KuvekPocketDescriptorCalculator which
142    # will perform the descriptor generation
143    descr_calc = Descr.KuvekPocketDescriptorCalculator()
144
145    descr_calc.sphereRadius = args.sphere_rad       # apply the -r argument
146    descr_calc.numTestVectors = args.num_test_vecs  # apply the -n argument
147
148    # modify the atom charge function to retrieve MMFF94 partial charges
149    descr_calc.atomChargeFunction = ForceField.getMMFF94Charge
150
151    # vector that will store the binding pocket descriptor values
152    descr = Math.DVector()
153
154    try:
155        descr_calc.calculate(args.sphere_ctr, rec_mol, descr)
156        
157        with open(args.out_file, 'w') as out_file:
158            for i in range(descr_calc.numTestVectors):
159                out_file.write(f'{descr(i * 2):.4f} {descr(i * 2 + 1):.4f}')
160                
161                if args.write_test_vecs:
162                    tv = descr_calc.getTestVector(i)
163                    out_file.write(f' {tv(0):.4f} {tv(1):.4f} {tv(2):.4f}')
164                        
165                out_file.write('\n')
166         
167    except Exception as e:    # handle exception raised in case of severe errors
168        sys.exit(f'Error: descriptor calculation failed:\n{str(e)}')
169   
170    if not args.quiet:
171        print('Done!')
172
173    sys.exit(0)
174        
175if __name__ == '__main__':
176    main()
