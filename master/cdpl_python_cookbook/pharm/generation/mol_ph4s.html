 2import CDPL.Pharm as Pharm

 5# generates the pharmacophore of the argument molecule using
 6# atom coordinates of the specified conformation
 7def genPharmacophore(mol: Chem.Molecule, conf_idx: int) -> Pharm.Pharmacophore:
 8    if conf_idx < 1:                                    # for a new molecule
 9        Pharm.prepareForPharmacophoreGeneration(mol)    # first call utility function preparing the molecule for pharmacophore generation
10        
11    ph4_gen = Pharm.DefaultPharmacophoreGenerator()     # create an instance of the pharmacophore generator default implementation
12    ph4 = Pharm.BasicPharmacophore()                    # create an instance of the default implementation of the Pharm.Pharmacophore interface
13    ph4_name = Chem.getName(mol)                        # use the name of the input molecule as pharmacophore name
14    
15    if conf_idx >= 0:                                   # if mol is a multi-conf. molecule use atom 3D coordinates of the specified conf.
16        ph4_gen.setAtom3DCoordinatesFunction(Chem.AtomConformer3DCoordinatesFunctor(conf_idx))
17        ph4_name += '#' + str(conf_idx)                 # and append conformer index to the pharmacophore name
18        
19    ph4_gen.generate(mol, ph4)          # generate the pharmacophore
20    Pharm.setName(ph4, ph4_name)        # set the pharmacophore name
21
22    return ph4
