 1import CDPL.Chem as Chem
 2import CDPL.ConfGen as ConfGen
 3
 4
 5# generates a low-energy 3D structure of the argument molecule using
 6# the provided initialized ConfGen.StructureGenerator instance
 7def gen3DStructure(mol: Chem.Molecule, struct_gen: ConfGen.StructureGenerator) -> int:
 8    # prepare the molecule for 3D structure generation
 9    ConfGen.prepareForConformerGeneration(mol) 
10
11    # generate the 3D structure
12    status = struct_gen.generate(mol)             
13    
14    # if sucessful, store the generated conformer ensemble as
15    # per atom 3D coordinates arrays (= the way conformers are represented in CDPKit)
16    if status == ConfGen.ReturnCode.SUCCESS:
17        struct_gen.setCoordinates(mol)                
18        
19    # return status code
20    return status
21
22def main() -> None:
23    args = parseArgs()
24    
25    # create reader for input molecules (format specified by file extension)
26    reader = Chem.MoleculeReader(args.in_file) 
27
28    # create writer for the generated 3D structures (format specified by file extension)
29    writer = Chem.MolecularGraphWriter(args.out_file) 
30
31    # export only a single 3D structure (in case of multi-conf. input molecules)
32    Chem.setMultiConfExportParameter(writer, False)
33    
34    # create and initialize an instance of the class ConfGen.StructureGenerator which will
35    # perform the actual 3D structure generation work
36    struct_gen = ConfGen.StructureGenerator()
37
38    struct_gen.settings.timeout = args.max_time * 1000 # apply the -t argument
39
40    # dictionary mapping status codes to human readable strings
41    status_to_str = { ConfGen.ReturnCode.UNINITIALIZED                  : 'uninitialized',
42                      ConfGen.ReturnCode.TIMEOUT                        : 'max. processing time exceeded',
43                      ConfGen.ReturnCode.ABORTED                        : 'aborted',
44                      ConfGen.ReturnCode.FORCEFIELD_SETUP_FAILED        : 'force field setup failed',
45                      ConfGen.ReturnCode.FORCEFIELD_MINIMIZATION_FAILED : 'force field structure refinement failed',
46                      ConfGen.ReturnCode.FRAGMENT_LIBRARY_NOT_SET       : 'fragment library not available',
47                      ConfGen.ReturnCode.FRAGMENT_CONF_GEN_FAILED       : 'fragment conformer generation failed',
48                      ConfGen.ReturnCode.FRAGMENT_CONF_GEN_TIMEOUT      : 'fragment conformer generation timeout',
49                      ConfGen.ReturnCode.FRAGMENT_ALREADY_PROCESSED     : 'fragment already processed',
50                      ConfGen.ReturnCode.TORSION_DRIVING_FAILED         : 'torsion driving failed',
51                      ConfGen.ReturnCode.CONF_GEN_FAILED                : 'conformer generation failed' }
52    
53    # create an instance of the default implementation of the Chem.Molecule interface
54    mol = Chem.BasicMolecule()
55    i = 1
56    
57    # read and process molecules one after the other until the end of input has been reached
58    try:
59        while reader.read(mol):
60            # compose a simple molecule identifier
61            mol_id = Chem.getName(mol).strip() 
62
63            if mol_id == '':
64                mol_id = '#' + str(i) # fallback if name is empty
65            else:
66                mol_id = '\'%s\' (#%s)' % (mol_id, str(i))
67
68            if not args.quiet:
69                print('- Generating 3D structure of molecule %s...' % mol_id)
70
71            try:
72                # generate 3D structure of the read molecule
73                status = gen3DStructure(mol, struct_gen) 
74
75                # check for severe error reported by status code
76                if status != ConfGen.ReturnCode.SUCCESS:
77                    if args.quiet:
78                        print('Error: 3D structure generation for molecule %s failed: %s' % (mol_id, status_to_str[status]))
79                    else:
80                        print(' -> 3D structure generation failed: %s' % status_to_str[status])
81                else: 
82                    # enforce the output of 3D coordinates in case of MDL file formats
83                    Chem.setMDLDimensionality(mol, 3)
84
85                    # output the generated 3D structure                    
86                    if not writer.write(mol):   
87                        sys.exit('Error: writing 3D structure of molecule %s failed' % mol_id)
88                        
89            except Exception as e:
90                sys.exit('Error: 3D structure generation or output for molecule %s failed: %s' % (mol_id, str(e)))
91
92            i += 1
93                
94    except Exception as e: # handle exception raised in case of severe read errors
95        sys.exit('Error: reading molecule failed: ' + str(e))
96
97    writer.close()
98    sys.exit(0)
