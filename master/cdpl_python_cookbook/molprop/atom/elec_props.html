 5# outputs the corresponding properties of each atom of the provided molecule
 6def outputProperties(mol: Chem.Molecule) -> None:
 7    Chem.calcImplicitHydrogenCounts(mol, False)  # calculate implicit hydrogen counts and set corresponding property for all atoms
 8    Chem.perceiveHybridizationStates(mol, False) # perceive atom hybridization states and set corresponding property for all atoms
 9    Chem.perceiveSSSR(mol, False)                # perceive smallest set of smallest rings and store as Chem.MolecularGraph property
10    Chem.setRingFlags(mol, False)                # perceive cycles and set corresponding atom and bond properties
11    Chem.setAromaticityFlags(mol, False)         # perceive aromaticity and set corresponding atom and bond properties
12    Chem.perceivePiElectronSystems(mol, False)   # perceive pi electron systems and store info as Chem.MolecularGraph property
13                                                 # (required for MHMO calculations)
14
15    # calculate sigma charges and electronegativities using the PEOE method and store values as atom properties
16    # (prerequisite for MHMO calculations)
17    MolProp.calcPEOEProperties(mol, False)  
18
19    # calculate pi charges, electronegativities and other properties by a modified Hueckel MO method and store values as properties
20    MolProp.calcMHMOProperties(mol, False)
21                
22    for atom in mol.atoms:
23        print('- Atom #%s' % str(atom.getIndex()))
24        print('\tSigma charge: %s' % str(MolProp.getPEOESigmaCharge(atom)))
25        print('\tPi charge: %s' % str(MolProp.getMHMOPiCharge(atom)))
26        print('\tTotal partial charge: %s' % str(MolProp.calcTotalPartialCharge(atom)))
27        print('\tLone-pair electronegativity: %s' % str(MolProp.calcLonePairElectronegativity(atom, mol)))
28        print('\tPi electronegativity: %s' % str(MolProp.calcPiElectronegativity(atom, mol)))
29        print('\tSigma electronegativity: %s' % str(MolProp.getPEOESigmaElectronegativity(atom)))
30        print('\tExerted inductive effect: %s' % str(MolProp.calcInductiveEffect(atom, mol)))
31        print('\tFree valence electron count: %s' % str(MolProp.calcFreeValenceElectronCount(atom, mol)))
32        print('\tValence electron count: %s' % str(MolProp.calcValenceElectronCount(atom)))
