 1import CDPL.Chem as Chem
 2import CDPL.MolProp as MolProp
 3
 4
 5# outputs the corresponding properties of each atom of the provided molecular graph
 6def outputProperties(molgraph: Chem.MolecularGraph) -> None:
 7    Chem.calcImplicitHydrogenCounts(molgraph, False)    # calculate implicit hydrogen counts and set corresponding property for all atoms
 8    Chem.perceiveHybridizationStates(molgraph, False)   # perceive atom hybridization states and set corresponding property for all atoms
 9    Chem.perceiveSSSR(molgraph, False)                  # perceive SSSR and store as Chem.MolecularGraph property
10    Chem.setRingFlags(molgraph, False)                  # perceive cycles and set corresponding atom and bond properties
11    Chem.setAromaticityFlags(molgraph, False)           # perceive aromaticity and set corresponding atom and bond properties
12    Chem.calcTopologicalDistanceMatrix(molgraph, False) # calculate topological distance matrix and store as Chem.MolecularGraph property
13                                                        # (required for effective polarizability calculations)
14    for atom in molgraph.atoms:
15        print('- Atom #%s' % str(molgraph.getAtomIndex(atom)))
16        print('\tHybrid polarizability: %s' % str(MolProp.getHybridPolarizability(atom, molgraph)))
17        print('\tEffective polarizability: %s' % str(MolProp.calcEffectivePolarizability(atom, molgraph)))
