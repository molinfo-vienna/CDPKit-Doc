 5# outputs the corresponding properties of each atom of the provided molecular graph
 6def outputProperties(molgraph: Chem.MolecularGraph) -> None:
 7    for atom in molgraph.atoms:
 8        print('- Atom #%s' % str(molgraph.getAtomIndex(atom)))
 9        print('\tAtomic weight: %s' % str(MolProp.getAtomicWeight(atom)))
10        print('\tPTE IUPAC group: %s' % str(MolProp.getIUPACGroup(atom)))
11        print('\tPTE period: %s' % str(MolProp.getPeriod(atom)))
12        print('\tVdW radius: %s' % str(MolProp.getVdWRadius(atom)))
13        print('\tCovalent radius (bond order=1): %s' % str(MolProp.getCovalentRadius(atom, 1)))
14        print('\tCovalent radius (bond order=2): %s' % str(MolProp.getCovalentRadius(atom, 2)))
15        print('\tCovalent radius (bond order=3): %s' % str(MolProp.getCovalentRadius(atom, 3)))
16        print('\tAllred Rochow electronegativity: %s' % str(MolProp.getAllredRochowElectronegativity(atom)))
17        print('\tElement name: %s' % MolProp.getElementName(atom))
18        print('\tValence electron count: %s' % str(MolProp.getElementValenceElectronCount(atom)))
19        print('\tAtom type specifies chemical element: %s' % str(MolProp.isChemicalElement(atom)))
20        print('\tIs main group element: %s' % str(MolProp.isMainGroupElement(atom)))
21        print('\tIs metal: %s' % str(MolProp.isMetal(atom)))
22        print('\tIs transition metal: %s' % str(MolProp.isTransitionMetal(atom)))
23        print('\tIs non-metal: %s' % str(MolProp.isNonMetal(atom)))
24        print('\tIs semi-metal: %s' % str(MolProp.isSemiMetal(atom)))
25        print('\tIs halogen: %s' % str(MolProp.isHalogen(atom)))
26        print('\tIs noble gas: %s' % str(MolProp.isNobleGas(atom)))
