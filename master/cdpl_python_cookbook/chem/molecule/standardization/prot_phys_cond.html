 4def processMolecules() -> None:
 5    args = parseArgs()
 6    
 7    # create reader for input molecules (format specified by file extension)
 8    reader = Chem.MoleculeReader(args.in_file) 
 9
10    # create writer for the output of matching molecules (format specified by file extension)
11    writer = Chem.MolecularGraphWriter(args.out_file) 
12
13    # create and initialize an instance of the class Chem.ProtonationStateStandardizer which
14    # implements the protonation state generation algorithm
15    prot_state_gen = Chem.ProtonationStateStandardizer()
16    
17    # create an instance of the default implementation of the Chem.Molecule interface
18    mol = Chem.BasicMolecule()
19    i = 1
20    
21    # read and process molecules one after the other until the end of input has been reached
22    try:
23        while reader.read(mol):
24            # compose a simple molecule identifier
25            mol_id = Chem.getName(mol).strip() 
26
27            if mol_id == '':
28                mol_id = '#' + str(i) # fallback if name is empty
29            else:
30                mol_id = '\'%s\' (#%s)' % (mol_id, str(i))
31
32            if not args.quiet:
33                print('- Processing molecule %s...' % mol_id)
34
35            try:
36                # protonate/deprotonate functional groups for phys. conditions
37                prot_state_gen.standardize(mol, Chem.ProtonationStateStandardizer.PHYSIOLOGICAL_CONDITION_STATE)
38
39                # enforce an update of the molecule components list (structure might have changed)
40                Chem.perceiveComponents(mol, True)
41                
42                # output the processed input molecule                 
43                if not writer.write(mol):   
44                    sys.exit('Error: output of molecule failed')
45                        
46            except Exception as e:
47                sys.exit('Error: processing or output of molecule %s failed: %s' % (mol_id, str(e)))
48
49            i += 1
50                
51    except Exception as e: # handle exception raised in case of severe read errors
52        sys.exit('Error: reading molecule failed: ' + str(e))
53
54    writer.close()
55    sys.exit(0)
