1.1.4.2. Rule-based Editing of SubstructuresÔÉÅ

The script edit_mols.py performs modifications on the molecular graph of molecules read from a specified file according to a provided set of SMARTS/SMILES pattern-based substructure editing rules and writes the results to a given output file.

Synopsis

python edit_mols.py [-h] -i <file> -o <file> -p <file/string> [-m] [-q]

Mandatory options

-i <file>

Molecule input file

-o <file>

Edited molecule output file

-p <file/string>

A string specifying search, (optional) exclude and result patterns or path to a file providing these (one set per line)

Other options

-h, --help

Show help message and exit

-m

Output input molecule before the resulting edited molecule (default: false)

-q

Disable progress output (default: false)

Examples

  • Nitro group standardization

    • Search Pattern: [#6][N:1](~[O:2])~[O:3]

    • Exclude Pattern: [#6][N+](=[O+0])-[O-]

    • Result Pattern: [~+:1](-[~-:2])=[~+0:3]

  8# exhaustively edits matching substructures of the argument molecule according to the 
  9# specified editing instructions using the provided list of initialized
 10# Chem.SubstructureEditor instances
 11def editMolecule(mol: Chem.Molecule, ed_list: list) -> int:
 12    # fill free valences with explicit hydrogens
 13    added_hs = Chem.makeHydrogenComplete(mol)  
 14
 15    # if expl. hydrogens were added clear the molecular graph components property (if present) since it became invalid
 16    if added_hs:                               
 17        Chem.clearComponents(mol)
 18 
 19    num_edits = 0
 20
 21    # perform the editing work via the provided Chem.SubstructureEditor instances
 22    for editor in ed_list:
 23        num_edits += editor.edit(mol)
 24
 25    # if structural changes were made clear 2D and 3D atom coordinates since they became invalid
 26    if num_edits > 0 or added_hs:
 27        Chem.setMDLDimensionality(mol, 0)        # for output in one of the MDL formats indicate that there are no atom coordinates present
 28
 29        for atom in mol.atoms:
 30            Chem.clear2DCoordinates(atom)        # delete 2D coordinates
 31            Chem.clear3DCoordinates(atom)        # delete 3D coordinates
 32            Chem.clear3DCoordinatesArray(atom)   # delete conformer ensemble coordinates 
 33
 34    return num_edits
 35
 36# creates and initializes a Chem.SubstructureEditor instance as specified
 37# by the given string of substructure search, exclude (optional) and editing result patterns
 38# in the format <#Search Patterns> <Search Pattern SMARTS> ...  <#Exclude Patterns> [<Exclude Pattern SMARTS> ...] <Result Pattern SMILES>
 39def createSubstructureEditor(ed_ptns: str) -> Chem.SubstructureEditor:
 40    editor = Chem.SubstructureEditor()
 41    tokens = ed_ptns.split()
 42    i = 1
 43
 44    for j in range(int(tokens[0])):
 45        editor.addSearchPattern(Chem.parseSMARTS(tokens[i]))
 46        i += 1
 47        
 48    for j in range(int(tokens[i])):
 49        i += 1
 50        editor.addExcludePattern(Chem.parseSMARTS(tokens[i]))
 51
 52    editor.setResultPattern(Chem.parseSMILES(tokens[i + 1]))
 53
 54    return editor
 55
 56# processes the value of the argument -p which is either the path to a file containing multiple lines
 57# of substructure editing specifications (one per line) or a string directly providing a complete set
 58# of search, exclude (optional) and result patterns (format is outline above)
 59def createSubstructureEditors(ed_ptns: str) -> list:
 60    if pathlib.Path(ed_ptns).is_file():  # if the argument value is a path to an existing file process it line by line
 61        editors = []
 62
 63        with open(ed_ptns, 'r') as ed_ptns_file:
 64            for line in ed_ptns_file.readlines():
 65                if line.startswith('#'): # lines starting with '#' ar comment lines
 66                    continue
 67
 68                editors.append(createSubstructureEditor(line))
 69
 70        return editors
 71
 72    # at this point the argument value directly specifies a set of search, exclude (optional) and result patterns
 73    return [ createSubstructureEditor(ed_ptns) ] 
 74
 75def parseArgs() -> argparse.Namespace:
 76    parser = argparse.ArgumentParser(description='Performs modifications on the molecular graph of molecules read from a \
 77    specified file according to a provided set of SMARTS/SMILES pattern-based substructure editing rules.')
 78
 79    parser.add_argument('-i',
 80                        dest='in_file',
 81                        required=True,
 82                        metavar='<file>',
 83                        help='Molecule input file')
 84    parser.add_argument('-o',
 85                        dest='out_file',
 86                        required=True,
 87                        metavar='<file>',
 88                        help='Edited molecule output file')
 89    parser.add_argument('-p',
 90                        dest='patterns',
 91                        required=True,
 92                        metavar='<file/string>',
 93                        help='A string specifying search, (optional) exclude and result patterns or path to a file providing these (one set per line)')
 94    parser.add_argument('-m',
 95                        dest='output_mol',
 96                        required=False,
 97                        action='store_true',
 98                        default=False,
 99                        help='Output input molecule before the resulting edited molecule (default: false)')
100    parser.add_argument('-q',
101                        dest='quiet',
102                        required=False,
103                        action='store_true',
104                        default=False,
105                        help='Disable progress output (default: false)')
106      
107    return parser.parse_args()
108
109def main() -> None:
110    args = parseArgs()
111    
112    # create reader for input molecules (format specified by file extension)
113    reader = Chem.MoleculeReader(args.in_file) 
114
115    # create writer for the generated 3D structures (format specified by file extension)
116    writer = Chem.MolecularGraphWriter(args.out_file) 
117
118    # create the list of one or more initialized Chem.SubstructureEditor instances doing the editing work
119    ed_list = createSubstructureEditors(args.patterns)
120    
121    # create an instance of the default implementation of the Chem.Molecule interface
122    mol = Chem.BasicMolecule()
123    i = 1
124    
125    # read and process molecules one after the other until the end of input has been reached
126    try:
127        while reader.read(mol):
128            # compose a simple molecule identifier
129            mol_id = Chem.getName(mol).strip() 
130
131            if mol_id == '':
132                mol_id = '#' + str(i) # fallback if name is empty
133            else:
134                mol_id = f'\'{mol_id}\' (#{i})'
135
136            try:
137                # output original molecule before the editing result
138                if args.output_mol:
139                    Chem.calcBasicProperties(mol, False)
140
141                    if not writer.write(mol):
142                        sys.exit(f'Error: writing molecule {mol_id} failed')
143
144                # modify the input molecule according to the specified editing rules 
145                num_changes = editMolecule(mol, ed_list)
146                
147                if not args.quiet:
148                    print(f'- Editing molecule {mol_id}: {num_changes} edit(s)')
149
150                Chem.calcBasicProperties(mol, False)
151                    
152                # output the edited molecule
153                if not writer.write(mol):   
154                    sys.exit(f'Error: writing edited molecule {mol_id} failed')
155                        
156            except Exception as e:
157                sys.exit(f'Error: editing or output of molecule {mol_id} failed: {str(e)}')
158
159            i += 1
160                
161    except Exception as e: # handle exception raised in case of severe read errors
162        sys.exit(f'Error: reading molecule failed: {str(e)}')
163
164    writer.close()
165    sys.exit(0)
166
167if __name__ == '__main__':
168    main()
