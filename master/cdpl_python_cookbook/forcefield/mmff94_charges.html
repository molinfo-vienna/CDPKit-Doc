 2import CDPL.ForceField as ForceField

 5# calculates and outputs the MMFF94 charges of the atoms of the provided molecular graph
 6def calcAndOutputCharges(mol: Chem.Molecule) -> None:
 7    Chem.calcImplicitHydrogenCounts(mol, False)  # calculate implicit hydrogen counts and set corresponding property for all atoms
 8    Chem.makeHydrogenComplete(mol)               # make all implicit hydrogens explicit
 9    Chem.perceiveHybridizationStates(mol, False) # perceive atom hybridization states and set corresponding property for all atoms
10    Chem.perceiveSSSR(mol, False)                # perceive smallest set of smallest rings and store as Chem.MolecularGraph property
11    Chem.setRingFlags(mol, False)                # perceive cycles and set corresponding atom and bond properties
12    Chem.setAromaticityFlags(mol, False)         # perceive aromaticity and set corresponding atom and bond properties
13  
14    ForceField.perceiveMMFF94AromaticRings(mol, False)        # perceive aromatic rings according to the MMFF94 aroamticity model and store data as Chem.MolecularGraph property
15    ForceField.assignMMFF94AtomTypes(mol, False, False)       # perceive MMFF94 atom types (tolerant mode) set corresponding property for all atoms
16    ForceField.assignMMFF94BondTypeIndices(mol, False, False) # perceive MMFF94 bond types (tolerant mode) set corresponding property for all bonds
17    ForceField.calcMMFF94AtomCharges(mol, False, False)       # calculate MMFF94 atom charges (tolerant mode) set corresponding property for all atoms
18
19    print('- MMFF94 partial charges')
20    
21    for atom in mol.atoms:
22        print('Atom #%s: %s' % (str(atom.getIndex()), str(ForceField.getMMFF94Charge(atom))))
