243 inline double CDPL::Descr::calcTanimotoSimilarity(const V& v1, const V& v2)
244 {
245  double ep12 = elemProd(v1, v2);
246 
247  return (ep12 / (elemProd(v1, v1) + elemProd(v2, v2) - ep12));
