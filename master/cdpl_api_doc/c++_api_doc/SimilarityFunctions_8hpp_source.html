295 template <typename V>
296 inline double CDPL::Descr::calcTanimotoSimilarity(const V& v1, const V& v2)
297 {
298  double ep12 = innerProd(v1, v2);
299 
300  return (ep12 / (innerProd(v1, v1) + innerProd(v2, v2) - ep12));
301 }
