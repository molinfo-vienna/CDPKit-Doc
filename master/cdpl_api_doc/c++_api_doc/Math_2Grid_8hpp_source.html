775  void resize(SizeType m, SizeType n, SizeType o, bool preserve = true, const ValueType& v = ValueType())
776  {
777  if (size1 == m && size2 == n && size3 == o)
778  return;
779 
780  if (preserve) {
781  Grid tmp(m, n, o, v);
782 
783  for (SizeType i = 0, min_size1 = std::min(size1, m); i < min_size1; i++)
784  for (SizeType j = 0, min_size2 = std::min(size2, n); j < min_size2; j++)
785  for (SizeType k = 0, min_size3 = std::min(size3, o); k < min_size3; k++)
786  tmp(i, j, k) = (*this)(i, j, k);
787 
788  swap(tmp);
789 
790  } else {
791  data.resize(storageSize(m, n, o), v);
792  size1 = m;
793  size2 = n;
794  size3 = o;
795  }
796  }
797 
798  private:
799  static SizeType storageSize(SizeType m, SizeType n, SizeType o)
800  {
801  CDPL_MATH_CHECK(n == 0 || o == 0 ||
802  (n <= (std::numeric_limits<SizeType>::max() / o) && m <= (std::numeric_limits<SizeType>::max() / (n * o))),
803  "Maximum size exceeded", Base::SizeError);
804  return (m * n * o);
805  }
