754 template <std::size_t Dim, typename T, typename Derived>
755 template <typename CoordsArray>
756 typename CDPL::Util::DGCoordinatesGeneratorBase<Dim, T, Derived>::ValueType
757 CDPL::Util::DGCoordinatesGeneratorBase<Dim, T, Derived>::getDistanceError(const CoordsArray& coords) const
758 {
759  ValueType error = ValueType();
760  ValueType pos_diff[Dim];
761 
762  for (typename DistanceConstraintList::const_iterator it = distConstraints.begin(), end = distConstraints.end(); it != end; ++it) {
763  const DistanceConstraint& constr = *it;
764 
765  ValueType dist_2 = calcDiffVectorAndSquaredDist(coords[constr.getPoint1Index()], coords[constr.getPoint2Index()], pos_diff);
766  ValueType dist = std::sqrt(dist_2);
767  ValueType lb = constr.getLowerBound();
768  ValueType ub = constr.getUpperBound();
769 
770  if (dist >= lb && dist <= ub)
771  continue;
772 
773  if (dist < lb) {
774  ValueType tmp = (dist_2 - lb * lb) / (0.000001 + lb * lb);
775 
776  error += tmp * tmp;
777 
778  } else {
779  ValueType tmp = (dist_2 - ub * ub) / (0.000001 + ub * ub);
780 
781  error += tmp * tmp;
782  }
783  }
784 
785  return error;
786 }

807 template <std::size_t Dim, typename T, typename Derived>
808 template <typename CoordsArray>
809 void CDPL::Util::DGCoordinatesGeneratorBase<Dim, T, Derived>::embedCoords(std::size_t num_points, CoordsArray& coords)
810 {
811  std::size_t num_dist_constrs = distConstraints.size();
812  std::size_t num_vol_constrs = static_cast<Derived&>(*this).getNumVolumeConstraints();
813 
814  if ((num_dist_constrs + num_vol_constrs) == 0)
815  return;
816 
817  std::size_t num_steps = std::size_t((num_dist_constrs + num_vol_constrs) * cycleStepCountFactor);
818  ValueType lambda = startLearningRate;
819 
820  if (num_dist_constrs > 0 && num_vol_constrs > 0) {
821  boost::random::uniform_int_distribution<std::size_t> constr_sd(0, num_dist_constrs + num_vol_constrs - 1);
822 
823  for (std::size_t i = 0; i < numCycles; i++, lambda -= learningRateDecr) {
824  for (std::size_t j = 0; j < num_steps; j++) {
825  std::size_t constr_idx = constr_sd(randomEngine);
826 
827  if (constr_idx < num_dist_constrs)
828  adjCoordsForDistanceConstraint(coords, lambda, constr_idx);
829  else
830  static_cast<Derived&>(*this).template adjCoordsForVolumeConstraint<CoordsArray>(coords, lambda, constr_idx - num_dist_constrs);
831  }
832  }
833 
834  return;
835  }
836 
837  if (num_dist_constrs > 0) {
838  boost::random::uniform_int_distribution<std::size_t> constr_sd(0, num_dist_constrs - 1);
839 
840  for (std::size_t i = 0; i < numCycles; i++, lambda -= learningRateDecr)
841  for (std::size_t j = 0; j < num_steps; j++)
842  adjCoordsForDistanceConstraint(coords, lambda, constr_sd(randomEngine));
843 
844  return;
845  }
846 
847  boost::random::uniform_int_distribution<std::size_t> constr_sd(0, num_vol_constrs - 1);
848 
849  for (std::size_t i = 0; i < numCycles; i++, lambda -= learningRateDecr)
850  for (std::size_t j = 0; j < num_steps; j++)
851  static_cast<Derived&>(*this).template adjCoordsForVolumeConstraint<CoordsArray>(coords, lambda, constr_sd(randomEngine));
852 }

864 template <std::size_t Dim, typename T, typename Derived>
865 template <typename Vec>
866 void CDPL::Util::DGCoordinatesGeneratorBase<Dim, T, Derived>::adjCoordsForConstraint(const DistanceConstraint& constr, Vec& pt1_pos,
867  Vec& pt2_pos, const ValueType& lambda) const
868 {
869  ValueType pos_diff[Dim];
870  ValueType dist = std::sqrt(calcDiffVectorAndSquaredDist(pt1_pos, pt2_pos, pos_diff));
871 
872  ValueType ub = constr.getUpperBound();
873  ValueType lb = constr.getLowerBound();
874 
875  if (dist >= lb && dist <= ub)
