520  template <typename T, typename C, typename GD, typename XF, typename V>
521  T interpolateTrilinear(const RegularSpatialGrid<T, C, GD, XF>& grid, const V& pos, bool local_pos)
522  {
523  typedef RegularSpatialGrid<T, C, GD, XF> GridType;
524 
525  typedef typename GridType::CoordinatesValueType CoordinatesValueType;
526  typedef typename GridType::SSizeType SSizeType;
527  typedef typename GridType::ValueType ValueType;
528 
529  if (grid.isEmpty())
530  return CoordinatesValueType();
531 
532  CoordinatesValueType loc_pos[3];
533 
534  if (local_pos) {
535  loc_pos[0] = pos[0];
536  loc_pos[1] = pos[1];
537  loc_pos[2] = pos[2];
538 
539  } else
540  grid.getLocalCoordinates(pos, loc_pos);
541 
542  SSizeType inds[3];
543 
544  grid.getLocalContainingCell(loc_pos, inds);
545 
546  CoordinatesValueType xyz0[3];
547 
548  grid.getLocalCoordinates(inds[0], inds[1], inds[2], xyz0);
549 
550  SSizeType inds_p1[3];
551 
552  inds_p1[0] = inds[0] + 1;
553  inds_p1[1] = inds[1] + 1;
554  inds_p1[2] = inds[2] + 1;
555 
556  inds[0] = std::max(SSizeType(0), inds[0]);
557  inds[1] = std::max(SSizeType(0), inds[1]);
558  inds[2] = std::max(SSizeType(0), inds[2]);
559 
560  inds[0] = std::min(SSizeType(grid.getSize1() - 1), inds[0]);
561  inds[1] = std::min(SSizeType(grid.getSize2() - 1), inds[1]);
562  inds[2] = std::min(SSizeType(grid.getSize3() - 1), inds[2]);
563 
564  inds_p1[0] = std::max(SSizeType(0), inds_p1[0]);
565  inds_p1[1] = std::max(SSizeType(0), inds_p1[1]);
566  inds_p1[2] = std::max(SSizeType(0), inds_p1[2]);
567 
568  inds_p1[0] = std::min(SSizeType(grid.getSize1() - 1), inds_p1[0]);
569  inds_p1[1] = std::min(SSizeType(grid.getSize2() - 1), inds_p1[1]);
570  inds_p1[2] = std::min(SSizeType(grid.getSize3() - 1), inds_p1[2]);
571 
572  CoordinatesValueType xd = (loc_pos[0] - xyz0[0]) / grid.getXStepSize();
573  CoordinatesValueType yd = (loc_pos[1] - xyz0[1]) / grid.getYStepSize();
574  CoordinatesValueType zd = (loc_pos[2] - xyz0[2]) / grid.getZStepSize();
575 
576  ValueType c00 = grid(inds[0], inds[1], inds[2]) * (1 - xd) + grid(inds_p1[0], inds[1], inds[2]) * xd;
577  ValueType c01 = grid(inds[0], inds[1], inds_p1[2]) * (1 - xd) + grid(inds_p1[0], inds[1], inds_p1[2]) * xd;
578  ValueType c10 = grid(inds[0], inds_p1[1], inds[2]) * (1 - xd) + grid(inds_p1[0], inds_p1[1], inds[2]) * xd;
579  ValueType c11 = grid(inds[0], inds_p1[1], inds_p1[2]) * (1 - xd) + grid(inds_p1[0], inds_p1[1], inds_p1[2]) * xd;
580 
581  ValueType c0 = c00 * (1 - yd) + c10 * yd;
582  ValueType c1 = c01 * (1 - yd) + c11 * yd;
583 
584  ValueType c = c0 * (1 - zd) + c1 * zd;
585 
586  return c;
587  }
