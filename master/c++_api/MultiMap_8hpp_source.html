766 template <typename Key, typename Value, bool AllowDefValues, typename KeyCompFunc>
767 Value& CDPL::Util::MultiMap<Key, Value, AllowDefValues, KeyCompFunc>::getValue(const Key& key)
768 {
769  EntryIterator lb = data.lower_bound(key);
770 
771  if (lb == data.end() || data.key_comp()(key, (*lb).first)) {
772  if (!AllowDefValues)
773  throw Base::ItemNotFound(std::string(getClassName()) + ": key not found");
774 
775  return (*data.insert(lb, Entry(key, getDefaultValue()))).second;
776  }
777 
778  return (*lb).second;
779 }

891 template <typename Key, typename Value, bool AllowDefValues, typename KeyCompFunc>
892 typename CDPL::Util::MultiMap<Key, Value, AllowDefValues, KeyCompFunc>::EntryIterator
893 CDPL::Util::MultiMap<Key, Value, AllowDefValues, KeyCompFunc>::setEntry(const Entry& item)
894 {
895  EntryIterator lb = data.lower_bound(item.first);
896  EntryIterator end = data.end();
897 
898  if (lb == end || data.key_comp()(item.first, (*lb).first))
899  return data.insert(lb, item);
900 
901  EntryIterator range_end = lb;
902 
903  while (range_end != end && !data.key_comp()(item.first, (*range_end).first))
904  ++range_end;
905 
906  data.erase(lb, range_end);
907 
908  return data.insert(range_end, item);
909 }
